import java.util.Scanner;
public class Main
{
	public static void main(String[] args) {
	    Scanner sc=new Scanner(System.in);
	    int n= sc.nextInt(),a,i;
	    boolean f;
	    for(a=2;a<=n;a++)
	    {
	        f=true;
	        for(i=2;i<=a/2;i++)
	        if(a%i==0)
	        {
	            f=false;
	            break;
	        }
	        if(f)
	        System.out.print(a+"_");
	    }
	}
}
